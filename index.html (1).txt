  const examDate = new Date("2026-03-15T09:00:00");
  function timer(){
    const now = new Date();
    const diff = examDate - now;
    if(diff<=0){
      document.getElementById('countdown').innerText = 'Exam Time';
      return;
    }
    const d = Math.floor(diff/(1000*60*60*24));
    const h = Math.floor((diff/(1000*60*60))%24);
    const m = Math.floor((diff/(1000*60))%60);
    const s = Math.floor((diff/1000)%60);
    document.getElementById('countdown').innerText = `${d}d ${h}h ${m}m ${s}s left`;
  }
  setInterval(timer,1000);

  let progress = localStorage.getItem('cuetProgress') || 0;
  function render(){
    document.getElementById('fill').style.width = progress+'%';
    document.getElementById('fill').innerText = progress+'%';
  }
  function increase(){
    if(progress<100){ progress++; }
    localStorage.setItem('cuetProgress',progress);
    render();
  }
  render();
</script></body>
